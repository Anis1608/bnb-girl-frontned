function confetti(){
  const cv=document.createElement('canvas');cv.style.cssText='position:fixed;inset:0;z-index:9999;pointer-events:none';
  cv.width=innerWidth;cv.height=innerHeight;document.body.appendChild(cv);
  const x=cv.getContext('2d');const cols=['#E8756A','#F4A07A','#E8924D','#22C55E','#D85A4E'];
  const P=Array.from({length:130},()=>({x:Math.random()*cv.width,y:-20,w:6+Math.random()*8,h:8+Math.random()*12,c:cols[Math.random()*cols.length|0],vx:(Math.random()-.5)*6,vy:3+Math.random()*4,r:Math.random()*6.28,vr:(Math.random()-.5)*.3,a:1}));
  let f=0;(function loop(){x.clearRect(0,0,cv.width,cv.height);let live=0;
    P.forEach(p=>{p.x+=p.vx;p.y+=p.vy;p.vy+=.08;p.r+=p.vr;if(p.y>cv.height*.82)p.a=Math.max(0,p.a-.03);
      if(p.a>0){live++;x.save();x.globalAlpha=p.a;x.translate(p.x,p.y);x.rotate(p.r);x.fillStyle=p.c;x.fillRect(-p.w/2,-p.h/2,p.w,p.h);x.restore();}});
    f++;if(live>0||f<40)requestAnimationFrame(loop);else document.body.removeChild(cv);})();
}

let curCat='',curQ='';

function runFilter(scroll){
  let n=0,first=null;
  document.querySelectorAll('.card').forEach(c=>{
    const ok=(!curCat||c.dataset.cat===curCat)&&(!curQ||c.dataset.search.includes(curQ));
    c.style.display=ok?'':'none';
    if(ok){c.classList.add('in');n++;if(!first)first=c;}
  });
  document.getElementById('count').textContent=n;
  document.getElementById('empty').style.display=n?'none':'block';
  if(scroll&&first){
    const off=document.querySelector('.life-strip').offsetHeight+parseInt(getComputedStyle(document.documentElement).getPropertyValue('--hdr'))+16;
    const y=document.getElementById('mentors').getBoundingClientRect().top+window.scrollY-off;
    window.scrollTo({top:y,behavior:'smooth'});
  }
}
function resetFilter(){curCat='';curQ='';document.querySelectorAll('.fch').forEach((c,i)=>c.classList.toggle('on',i===0));const s=document.getElementById('search');if(s)s.value='';runFilter(false);}
